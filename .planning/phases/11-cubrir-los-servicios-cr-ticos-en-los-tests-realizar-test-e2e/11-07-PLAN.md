    ```python
    import pytest
    import subprocess
    import re
    
    @pytest.mark.e2e
    class TestLucienVoiceConsistency:
        def test_no_hardcoded_spanish_in_services(self):
            """
            Grep services/ and handlers/ for hardcoded Spanish user-facing strings
            that should live in utils/lucien_voice.py instead.
            """
            result = subprocess.run(
                [
                    "grep", "-r", "-n", "--include=*.py",
                    "-E", r"[A-ZÁÉÍÓÚÑ][a-záéíóúñ]+.*((El|La|Los|Las|Su|Ha|No|Si|¿|¡|Diana|Lucien|besitos|regalo|promoción|paquete|misión|recompensa|acceso|VIP|canal|gracias|bienvenido|error|correctamente|disponible|encontrado|agotado))",
                    "services/", "handlers/"
                ],
                capture_output=True, text=True
            )
    
            lines = result.stdout.strip().split("\n") if result.stdout.strip() else []
            filtered = []
            for line in lines:
                if not line:
                    continue
                # Skip lucien_voice.py (the canonical message source)
                if "lucien_voice" in line:
                    continue
                # Skip logger calls and log messages
                if "logger" in line or "logging" in line or "f\"" in line and "log" in line.lower():
                    continue
                # Skip class/function definitions
                if re.search(r"^\s*(class\s+|def\s+|@pytest)", line.split(":", 2)[-1] if ":" in line else ""):
                    continue
                # Skip comments
                if re.search(r"^\s*#", line.split(":", 2)[-1] if ":" in line else ""):
                    continue
                # Skip docstrings / triple quotes
                if '"""' in line or "'''" in line:
                    continue
                # Skip import lines
                if re.search(r"^\s*(from\s+|import\s+)", line.split(":", 2)[-1] if ":" in line else ""):
                    continue
                filtered.append(line)
    
            assert len(filtered) == 0, (
                "Hardcoded Spanish user-facing strings found outside message templates:\n"
                + "\n".join(filtered[:20])
            )
    
        def test_lucien_voice_file_exists_and_exports(self):
            """Verify that the central voice file exists and is importable."""
            from utils.lucien_voice import LucienVoice
            # Verify it's a class or module with expected attributes
            assert LucienVoice is not None
    ```

    Note: The regex may need tuning. The executor should run the test, inspect failures, and adjust the regex or exclusions if there are false positives (e.g., legitimate variable names, model class docstrings). The goal is to catch actual user-facing message strings hardcoded in services/handlers. If the grep produces false positives, refine the pattern iteratively until the test passes cleanly.
