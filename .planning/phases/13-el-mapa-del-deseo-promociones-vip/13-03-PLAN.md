def create_vip_promotions():
    """Crea las 3 promociones VIP si no existen"""
    promotion_service = PromotionService()
    package_service = PackageService()

    try:
        # Verificar si ya existen promociones VIP
        existing = promotion_service.get_vip_exclusive_promotions()
        if existing:
            logger.info(f"Ya existen {len(existing)} promociones VIP. Saltando...")
            for promo in existing:
                logger.info(f"  - {promo.name}: {promo.price_display}")
            return

        # Crear un paquete dummy para las promociones VIP si no existe
        packages = package_service.get_all_packages()
        if packages:
            package_id = packages[0].id
            logger.info(f"Usando paquete existente ID={package_id}")
        else:
            logger.error("No hay paquetes disponibles. Cree un paquete primero.")
            return

        # Crear las 3 promociones
        for promo_data in VIP_PROMOTIONS:
            promo = promotion_service.create_promotion(
                name=promo_data["name"],
                description=promo_data["description"],
                package_id=package_id,
                price_mxn=promo_data["price_mxn"],
                created_by=None  # Sistema
            )

            # Actualizar para marcar como VIP exclusiva
            promotion_service.update_promotion(
                promo.id,
                is_vip_exclusive=True
            )

            logger.info(f"Creada promoción VIP: {promo.name} - {promo.price_display}")

        logger.info("✓ Las 3 promociones VIP del Mapa del Deseo han sido creadas")

    finally:
        promotion_service.close()
        package_service.close()


if __name__ == "__main__":

    create_vip_promotions()
